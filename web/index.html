<script>

const API='https://afl-fantasy-analytics.onrender.com';

const BUDGET=14130000;

const SLOTS={DEF:8,MID:8,RUC:2,FWD:8};

const BENCH=4;

const TOTAL_SLOTS=Object.values(SLOTS).reduce((a,b)=>a+b,0)+BENCH;


const TEAM_NAMES={

  WBD:'Western Bulldogs',WCE:'West Coast Eagles',ADE:'Adelaide Crows',

  BRI:'Brisbane Lions',MEL:'Melbourne Demons',NTH:'North Melbourne',

  GEE:'Geelong Cats',GCS:'Gold Coast Suns',GWS:'GWS Giants',

  HAW:'Hawthorn Hawks',CAR:'Carlton Blues',COL:'Collingwood Magpies',

  STK:'St Kilda Saints',SYD:'Sydney Swans',PTA:'Port Adelaide Power',

  RIC:'Richmond Tigers',ESS:'Essendon Bombers',FRE:'Fremantle Dockers'

};


const POS_COLORS={DEF:'var(--def)',MID:'var(--mid)',RUC:'var(--ruc)',FWD:'var(--fwd)'};


let allPlayers=[];

let myTeam=[];

let curPos='ALL';

let curSort='predicted_points';


/* ── fetch ── */

async function loadPlayers(){

  try{

    const r=await fetch(`${API}/players`);

    if(!r.ok) throw new Error(r.status);

    allPlayers=await r.json();

    buildTeamFilter();

    renderList();

    renderTeam();

  }catch(e){

    document.getElementById('player-list').innerHTML=

      `<div class="loading" style="color:#ff5252">

        <div class="empty-icon">!</div>

        <div>Could not reach API</div>

        <div style="font-size:11px;margin-top:4px">${e.message}</div>

      </div>`;

  }

}


function teamAbbr(name){return name.split('-')[0]}


function buildTeamFilter(){

  const abbrs=[...new Set(allPlayers.map(p=>teamAbbr(p.name)))].sort();

  const sel=document.getElementById('team-filter');

  abbrs.forEach(a=>{

    const o=document.createElement('option');

    o.value=a;o.textContent=TEAM_NAMES[a]||a;

    sel.appendChild(o);

  });

}


/* ── filter + sort ── */

function visiblePlayers(){

  const q=document.getElementById('search').value.toLowerCase().trim();

  const team=document.getElementById('team-filter').value;

  return allPlayers

    .filter(p=>{

      if(curPos!=='ALL'&&p.position!==curPos)return false;

      if(team&&!p.name.startsWith(team))return false;

      if(q&&!p.name.toLowerCase().includes(q)&&!(TEAM_NAMES[teamAbbr(p.name)]||'').toLowerCase().includes(q))return false;

      return true;

    })

    .sort((a,b)=>b[curSort]-a[curSort]);

}


/* ── formatting ── */

const $m=v=>'$'+Math.round(v).toLocaleString();

const $p=v=>Math.round(v);

function vClass(v){return v>=30?'hi':v>=22?'md':'lo'}


function inTeam(id){return myTeam.some(p=>p.player_id===id)}


/* ── render player list ── */

function renderList(){

  const pl=document.getElementById('player-list');

  const players=visiblePlayers();

  document.getElementById('result-count').textContent=`${players.length} player${players.length!==1?'s':''}`;


  if(!players.length){

    pl.innerHTML=`<div class="empty-state"><div class="empty-icon">&#127944;</div><div>No players match</div></div>`;

    return;

  }


  const maxH=Math.max(...players.map(p=>p.ci_high));

  pl.innerHTML='';


  players.forEach((p,i)=>{

    const abbr=teamAbbr(p.name);

    const used=inTeam(p.player_id);

    const pct=v=>((v/maxH)*100).toFixed(2);

    const fillL=pct(p.ci_low);

    const fillW=pct(p.ci_high-p.ci_low);

    const dotP=pct(p.predicted_points);


    const card=document.createElement('div');

    card.className='pcard'+(used?' used':'');

    card.style.animationDelay=`${Math.min(i*0.018,0.4)}s`;

    card.innerHTML=`

      <div class="stripe ${p.position}"></div>

      <div class="pinfo">

        <div class="pname-row">

          <div class="pname">${p.name}</div>

          <span class="pbadge ${p.position}">${p.position}</span>

          <span class="pteam">${abbr}</span>

        </div>

        <div class="pstats">

          <div class="sitem">

            <div class="sval">${$p(p.predicted_points)}</div>

            <div class="slbl">Proj Pts</div>

          </div>

          <div class="sitem">

            <div class="sval vprice">${$m(p.price)}</div>

            <div class="slbl">Price</div>

          </div>

          <div class="sitem">

            <span class="vchip ${vClass(p.value_vs_price)}">${p.value_vs_price.toFixed(1)}</span>

            <div class="slbl" style="margin-top:3px">Value</div>

          </div>

          <div class="ci-wrap">

            <div class="ci-bar" title="${$p(p.ci_low)} – ${$p(p.ci_high)} projected range">

              <div class="ci-fill" style="left:${fillL}%;width:${fillW}%"></div>

              <div class="ci-dot" style="left:${dotP}%"></div>

            </div>

            <div class="ci-range">${$p(p.ci_low)} – ${$p(p.ci_high)} range</div>

          </div>

        </div>

      </div>

      <button class="add-btn" title="Add to team" onclick="addPlayer(${p.player_id})">+</button>

    `;

    pl.appendChild(card);

  });

}


/* ── add / remove ── */

function addPlayer(id){

  const p=allPlayers.find(x=>x.player_id===id);

  if(!p)return;

  if(inTeam(id)){toast('Already in squad','err');return}

  if(myTeam.length>=TOTAL_SLOTS){toast(`Squad full (${TOTAL_SLOTS})`, 'err');return}

  const posCount=myTeam.filter(x=>x.position===p.position).length;

  if(posCount>=SLOTS[p.position]){toast(`${p.position} slots full (${SLOTS[p.position]} max)`,'err');return}

  const spent=myTeam.reduce((a,x)=>a+x.price,0);

  if(spent+p.price>BUDGET){toast('Exceeds budget!','err');return}

  myTeam.push(p);

  toast(`${p.name} added!`,'ok');

  sync();

}


function removePlayer(id){

  myTeam=myTeam.filter(p=>p.player_id!==id);

  sync();

}


function sync(){

  updateHeader();

  renderTeam();

  renderList();

}


/* ── header ── */

function updateHeader(){

  const spent=myTeam.reduce((a,p)=>a+p.price,0);

  const rem=BUDGET-spent;

  const pts=myTeam.reduce((a,p)=>a+p.predicted_points,0);

  document.getElementById('hdr-count').textContent=`${myTeam.length} / ${TOTAL_SLOTS}`;

  const hb=document.getElementById('hdr-budget');

  hb.textContent=$m(rem);

  hb.className='hdr-pill-val '+(rem>3e6?'ok':rem>1e6?'warn':'danger');

  document.getElementById('hdr-pts').textContent=myTeam.length?Math.round(pts):'—';

}


/* ── render team ── */

function renderTeam(){

  const spent=myTeam.reduce((a,p)=>a+p.price,0);

  const rem=BUDGET-spent;

  const usedPct=(spent/BUDGET*100).toFixed(2);

  const pts=myTeam.reduce((a,p)=>a+p.predicted_points,0);

  const avgVal=myTeam.length?myTeam.reduce((a,p)=>a+p.value_vs_price,0)/myTeam.length:0;


  const bv=document.getElementById('bval');

  bv.textContent=$m(rem);

  bv.className='bval '+(rem>3e6?'ok':rem>1e6?'warn':'danger');


  const bf=document.getElementById('bfill');

  bf.style.width=`${100-usedPct}%`;

  bf.style.background=rem>3e6?'#3ee88e':rem>1e6?'#ff9f43':'#ff5252';


  document.getElementById('mc-count').textContent=myTeam.length;

  document.getElementById('mc-pts').textContent=myTeam.length?Math.round(pts):'—';

  document.getElementById('mc-val').textContent=myTeam.length?avgVal.toFixed(1):'—';


  const tbody=document.getElementById('tbody');


  if(myTeam.length===0){

    tbody.innerHTML=`

      <div class="empty-state" style="height:180px">

        <div class="empty-icon">&#127944;</div>

        <div>Your squad is empty</div>

        <div style="font-size:11px;margin-top:4px;color:var(--text-muted)">Click + on any player to add</div>

      </div>

    `;

    return;

  }


  let html='';

  Object.entries(SLOTS).forEach(([pos,max])=>{

    const pp=myTeam.filter(p=>p.position===pos);

    html+=`

      <div class="psect">

        <div class="psect-hdr">

          <div class="psect-title" style="color:${POS_COLORS[pos]}">${pos}</div>

          <div class="psect-count">${pp.length} / ${max}</div>

        </div>

    `;

    for(let i=0;i<max;i++){

      const p=pp[i];

      if(p){

        html+=`

          <div class="tslot filled">

            <div class="sn" title="${p.name}">${p.name}</div>

            <div class="sp">${$m(p.price)}</div>

            <div class="spt">${$p(p.predicted_points)}</div>

            <button class="rmbtn" onclick="removePlayer(${p.player_id})" title="Remove">×</button>

          </div>`;

      }else{

        html+=`<div class="tslot"><span>Empty slot</span></div>`;

      }

    }

    html+='</div>';

  });


  tbody.innerHTML=html;

}
